def generate_modern_shop_layout(product_list_items): """ Generiert HTML-Code für ein modernes und gut strukturiertes Shop-Layout mit einer Produktliste. Args: product_list_items (list): Eine Liste von Strings, die die einzelnen Produkte repräsentieren. Jeder String sollte idealerweise Produktname, Preis und eine kurze Beschreibung enthalten. Returns: str: HTML-Code für das Shop-Layout. """ # Grundstruktur des HTML-Dokuments html_content = """ Moderner Shop

Willkommen in unserem modernen Shop!

Entdecken Sie unsere sorgfältig ausgewählten Produkte.

""" # Hinzufügen der einzelnen Produkt-Elemente for item in product_list_items: # Hier wird angenommen, dass `item` bereits gut formatiert ist. # Für eine komplexere Anwendung könnte man hier eine separate Funktion # zur Aufbereitung der Produktinformationen aufrufen. product_name = f"Produkt: {item.split(',')[0].strip()}" if ',' in item else item product_price = f"Preis: {item.split(',')[1].strip()}" if ',' in item and len(item.split(',')) > 1 else "Preis nicht verfügbar" product_description = f"Beschreibung: {item.split(',')[2].strip()}" if ',' in item and len(item.split(',')) > 2 else "Eine kurze Beschreibung unseres tollen Produkts." html_content += f"""

{product_name}

{product_price}

{product_description}

""" # Abschluss des HTML-Dokuments html_content += """

© 2023 Ihr Moderner Shop. Alle Rechte vorbehalten.

""" return html_content # Beispielhafte Verwendung: # product_data = [ # "Laptop, 1200€, Ein leistungsstarker Laptop für Profis.", # "Smartphone, 750€, Das neueste Modell mit beeindruckender Kamera.", # "Kopfhörer, 150€, Kristallklarer Sound für unterwegs.", # "Smartwatch, 200€, Ihr intelligenter Begleiter für den Alltag." # ] # # shop_html = generate_modern_shop_layout(product_data) # print(shop_html)
